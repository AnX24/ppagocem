CREATE TABLE ADMIN (
  IDADMIN INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  IDPERSONA INTEGER UNSIGNED NOT NULL,
  IDUSUARIO VARCHAR(25) NOT NULL,
  CORREOADMIN VARCHAR(55) NULL,
  PRIMARY KEY(IDADMIN),
  INDEX ADMIN_FKIndex1(IDUSUARIO),
  INDEX ADMIN_FKIndex2(IDPERSONA)
);

CREATE TABLE BANCO (
  IDBANCO INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  NOMBRE VARCHAR(50) NULL,
  PRIMARY KEY(IDBANCO)
);

CREATE TABLE CLIENTE (
  IDCLIENTE INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  IDPERSONA INTEGER UNSIGNED NOT NULL,
  DIRECCION VARCHAR(250) NULL,
  FONODIRECCION VARCHAR(13) NULL,
  COMUNA VARCHAR(50) NULL,
  PRIMARY KEY(IDCLIENTE),
  INDEX CLIENTE_FKIndex1(IDPERSONA)
);

CREATE TABLE COBRADOR (
  IDCOBRADOR INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  IDPERSONA INTEGER UNSIGNED NOT NULL,
  IDUSUARIO VARCHAR(25) NOT NULL,
  CORREOEMPLEADO VARCHAR(55) NULL,
  BLOQUEO BOOL NULL,
  PRIMARY KEY(IDCOBRADOR),
  INDEX COBRADOR_FKIndex1(IDUSUARIO),
  INDEX COBRADOR_FKIndex2(IDPERSONA)
);

CREATE TABLE DETALLECHEQUE (
  IDDETALLECHEQUE INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  IDDETALLEPAGO INTEGER UNSIGNED NOT NULL,
  IDBANCO INTEGER UNSIGNED NOT NULL,
  MONTO INTEGER UNSIGNED NULL,
  NOMBREORIGEN VARCHAR(50) NULL,
  NOMBREDESTINO VARCHAR(50) NULL,
  FECHACOBRO DATE NULL,
  DOCUMENTO VARCHAR(255) NULL,
  PRIMARY KEY(IDDETALLECHEQUE),
  INDEX DETALLECHEQUE_FKIndex1(IDBANCO),
  INDEX DETALLECHEQUE_FKIndex2(IDDETALLEPAGO)
);

CREATE TABLE DETALLEPAGO (
  IDDETALLEPAGO INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  IDPAGO INTEGER UNSIGNED NOT NULL,
  IDTIPOTRANSACCION INTEGER UNSIGNED NOT NULL,
  FECHATRANSACCION DATETIME NULL,
  PRIMARY KEY(IDDETALLEPAGO),
  INDEX DETALLEPAGO_FKIndex1(IDTIPOTRANSACCION),
  INDEX DETALLEPAGO_FKIndex2(IDPAGO)
);

CREATE TABLE DETALLETAS (
  IDDETALLETAS INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  IDTAS INTEGER UNSIGNED NOT NULL,
  IDDETALLEPAGO INTEGER UNSIGNED NOT NULL,
  MONTO INTEGER UNSIGNED NULL,
  BOLETA VARCHAR(255) NULL,
  PRIMARY KEY(IDDETALLETAS),
  INDEX DETALLETAS_FKIndex1(IDDETALLEPAGO),
  INDEX DETALLETAS_FKIndex2(IDTAS)
);

CREATE TABLE DETEXT (
  IDDETEXT INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  IDTAS INTEGER UNSIGNED NOT NULL,
  DOCUMENTO VARCHAR(255) NULL,
  PRIMARY KEY(IDDETEXT),
  INDEX DETEXT_FKIndex1(IDTAS)
);

CREATE TABLE DETTRANSACCION (
  IDDETTRANSACCION INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  IDDETALLEPAGO INTEGER UNSIGNED NOT NULL,
  MONTOTRANSACCION INTEGER UNSIGNED NULL,
  BOLETAELECTRONICA VARCHAR(255) NULL,
  PRIMARY KEY(IDDETTRANSACCION),
  INDEX DETTRANSACCION_FKIndex1(IDDETALLEPAGO)
);

CREATE TABLE DUMBAR (
  IDDUMBAR INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  IDPERSONA INTEGER UNSIGNED NOT NULL,
  IDUSUARIO VARCHAR(25) NOT NULL,
  ACCESODUMBAR BOOL NULL,
  CORREO_DUMBAR INTEGER UNSIGNED NULL,
  PRIMARY KEY(IDDUMBAR),
  INDEX DUMBAR_FKIndex1(IDUSUARIO),
  INDEX DUMBAR_FKIndex2(IDPERSONA)
);

CREATE TABLE PAGO (
  IDPAGO INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  IDCLIENTE INTEGER UNSIGNED NOT NULL,
  IDRUTA INTEGER UNSIGNED NOT NULL,
  FECHAPAGO DATETIME NULL,
  VALORPAGO INTEGER UNSIGNED NULL,
  PRIMARY KEY(IDPAGO),
  INDEX PAGO_FKIndex1(IDRUTA),
  INDEX PAGO_FKIndex2(IDCLIENTE)
);

CREATE TABLE PERSONA (
  IDPERSONA INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  NOMBRE VARCHAR(25) NULL,
  AP_P VARCHAR(25) NULL,
  AP_M VARCHAR(25) NULL,
  PRIMARY KEY(IDPERSONA)
);

CREATE TABLE RUTA (
  IDRUTA INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  IDCOBRADOR INTEGER UNSIGNED NOT NULL,
  NUMEROVISITAS INTEGER UNSIGNED NULL,
  PRIMARY KEY(IDRUTA),
  INDEX RUTA_FKIndex1(IDCOBRADOR)
);

CREATE TABLE TAS (
  IDTAS INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  UBICACION VARCHAR(50) NULL,
  ESTADO INTEGER UNSIGNED NULL,
  PRIMARY KEY(IDTAS)
);

CREATE TABLE TIPOTRANSACCION (
  IDTIPOTRANSACCION INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  NOMBRE VARCHAR(25) NULL,
  PRIMARY KEY(IDTIPOTRANSACCION)
);

CREATE TABLE USUARIO (
  IDUSUARIO VARCHAR(25) NOT NULL AUTO_INCREMENT,
  CONTRASENA VARCHAR(25) NOT NULL,
  PRIMARY KEY(IDUSUARIO)
);
